    <script>
        // === رابط تحميل APK الخاص بك ===
        const APK_URL = "https://github.com/mfoad2058-star/rolefy-landing/releases/download/v1.0/app-release.apk";

        // === إعدادات العداد ===
        const NAMESPACE = "rolefy-app";
        const KEY = "downloads";

        const COUNTER_API = `https://api.countapi.xyz/hit/${NAMESPACE}/${KEY}`;
        const downloadBtn = document.getElementById('downloadBtn');
        const downloadCounter = document.getElementById('downloadCounter');

        // ربط الزر برابط التحميل
        downloadBtn.href = APK_URL;

        // جلب العدد الحالي
        async function fetchCount() {
            try {
                const response = await fetch(`https://api.countapi.xyz/get/${NAMESPACE}/${KEY}`);
                const data = await response.json();
                if (data && data.value !== undefined) {
                    downloadCounter.textContent = data.value;
                } else {
                    downloadCounter.textContent = '0';
                }
            } catch (err) {
                downloadCounter.textContent = '...';
            }
        }

        // زيادة العداد عند النقر على التحميل
        downloadBtn.addEventListener('click', async () => {
            try {
                const response = await fetch(COUNTER_API);
                const data = await response.json();
                if (data && data.value !== undefined) {
                    downloadCounter.textContent = data.value;
                }
            } catch (err) {
                // حتى لو فشل العداد، التحميل سيستمر
            }
        });

        // عرض العدد عند تحميل الصفحة
        fetchCount();
    </script>
